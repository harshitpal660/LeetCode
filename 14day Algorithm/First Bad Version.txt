class Solution:
    def firstBadVersion(self, n: int) -> int:
        p=1;
        q=n;
        
        while(p<q):
            mid=p+(q-p)/2;
            if(isBadVersion(mid)):
                q = mid
            else:
                p = mid+1
        return int(p)
